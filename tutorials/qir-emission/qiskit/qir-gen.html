def teleportation(qc):
    qc.h(q[1])
    qc.cx(q[1], q[2])
    qc.cx(q[0], q[1])
    qc.h(q[0])
    qc.measure(q[0], c0[0])
    qc.measure(q[1], c1[0])
    qc.z(q[2]).c_if(c0, 1)
    qc.x(q[2]).c_if(c1, 1)
    qc.measure(q[2], c2[0])
